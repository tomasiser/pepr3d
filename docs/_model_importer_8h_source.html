78  static bool zeroAreaCheck(const std::array<glm::vec3, 3> &triangle, const double Eps = 0.000001) {
80  const double len = glm::length(glm::cross(triangle[1] - triangle[0], triangle[2] - triangle[0]));
81  const bool hasZeroAreaByCross = glm::epsilonEqual<double>(len, 0, Eps);
82  const bool verticesAreDifferent =
83  triangle[0] != triangle[1] && triangle[0] != triangle[2] && triangle[1] != triangle[2];
84  if(verticesAreDifferent && !hasZeroAreaByCross) {
85  return false;
86  } else {
87  return true;
88  }
89  }

104  std::vector<std::array<size_t, 3>> indices;
105  indices.reserve(mesh->mNumFaces);
106 
107  for(unsigned int i = 0; i < mesh->mNumFaces; i++) {
108  P_ASSERT(mesh->mFaces[i].mNumIndices == 3);
109

111 
112  for(unsigned int j = 0; j < mesh->mFaces[i].mNumIndices; j++) {
113  triangle[j].x = mesh->mVertices[mesh->mFaces[i].mIndices[j]].x;
114  triangle[j].y = mesh->mVertices[mesh->mFaces[i].mIndices[j]].y;
115  triangle[j].z = mesh->mVertices[mesh->mFaces[i].mIndices[j]].z;
116  }
117 
119  const bool isZeroArea = zeroAreaCheck(triangle);
120  if(!isZeroArea) {
121  indices.push_back(

234  std::vector<DataTriangle> processFirstMesh(aiMesh *mesh) {
235  std::vector<DataTriangle> triangles;
236 
238  std::unordered_map<std::array<float, 3>, size_t, boost::hash<std::array<float, 3>>> colorLookup;
239 
240  for(unsigned int i = 0; i < mesh->mNumFaces; i++) {
241  aiFace face = mesh->mFaces[i];
242 
243  P_ASSERT(face.mNumIndices == 3);
244 
245  std::array<glm::vec3, 3> vertices;
246  glm::vec3 normals[3];
247  glm::vec3 normal;
248 
250  for(unsigned int j = 0; j < face.mNumIndices; j++) {
251  vertices[j].x = mesh->mVertices[face.mIndices[j]].x;
252  vertices[j].y = mesh->mVertices[face.mIndices[j]].y;
253  vertices[j].z = mesh->mVertices[face.mIndices[j]].z;
254 
255  if(mesh->HasNormals()) {
256  normals[j].x = mesh->mNormals[face.mIndices[j]].x;
257  normals[j].y = mesh->mNormals[face.mIndices[j]].y;
258  normals[j].z = mesh->mNormals[face.mIndices[j]].z;
259  }
260  }
261 
263  normal = calculateNormal(vertices, normals);
264 
265  glm::vec4 color;
266  size_t returnColor = 0;
267  if(mesh->GetNumColorChannels() > 0) {
268  color.r = mesh->mColors[0][face.mIndices[0]][0]; // first color layer from first vertex of triangle
269  color.g = mesh->mColors[0][face.mIndices[0]][1];
270  color.b = mesh->mColors[0][face.mIndices[0]][2];
271  color.a = 1;
272 
273  const std::array<float, 3> rgbArray = {color.r, color.g, color.b};
274  const auto result = colorLookup.find(rgbArray);
275  if(result != colorLookup.end()) {
276  P_ASSERT(result->second < mPalette.size());
277  P_ASSERT(result->second >= 0);
278  returnColor = result->second;
279  } else {
280  mPalette.addColor(color);
281  colorLookup.insert({rgbArray, mPalette.size() - 1});
282  returnColor = mPalette.size() - 1;
283  P_ASSERT(colorLookup.find(rgbArray) != colorLookup.end());
284  }
285  }
286 
288  const double Eps = 0.000001;
289  const bool isZeroArea = zeroAreaCheck(vertices, Eps);
290  if(!isZeroArea) {
292  // Normal should be normalized
293  P_ASSERT(glm::epsilonEqual<double>(glm::length(normal), 1.0, Eps));
294  // ColorPalette should either be empty and return color 0, or returnColor should be within the palette
295  P_ASSERT(
296  (mPalette.size() == 0 && returnColor == 0) ||
297  (mPalette.size() > 0 && returnColor < mPalette.size() && returnColor < PEPR3D_MAX_PALETTE_COLORS));
299  triangles.emplace_back(vertices[0], vertices[1], vertices[2], normal, returnColor);
300  } else {
301  CI_LOG_W("Imported a triangle with zero surface area. Ommiting it from geometry data.");
302  }
303  }
304  return triangles;
305  }

308  static glm::vec3 calculateNormal(const std::array<glm::vec3, 3> vertices, const glm::vec3 normals[3]) {
309  const glm::vec3 p0 = vertices[1] - vertices[0];
310  const glm::vec3 p1 = vertices[2] - vertices[0];
311  const glm::vec3 faceNormal = glm::normalize(glm::cross(p0, p1));
312 
313  auto areNormalsNan = glm::isnan(normals[0]);
314  if(glm::any(areNormalsNan)) {
315  return faceNormal;
316  }
317 
318  const glm::vec3 vertexNormal = glm::normalize(normals[0] + normals[1] + normals[2]);
319  const float dot = glm::dot(faceNormal, vertexNormal);
320 
321  // return vertexNormal;
322  return (dot < 0.0f) ? -faceNormal : faceNormal;
323  }
