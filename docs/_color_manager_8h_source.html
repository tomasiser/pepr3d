128  static void generateColors(const size_t colorCount, std::vector<glm::vec4>& outNewColors) {
129  outNewColors.clear();
130  outNewColors.reserve(colorCount);
131 
132  std::random_device rd; // Will be used to obtain a seed for the random number engine
133  std::mt19937 gen(rd()); // Standard mersenne_twister_engine seeded with rd()
134  std::uniform_real_distribution<> randomGenHue(0.0, 1.0);
135  std::uniform_real_distribution<> randomGenValue(0.6, 1.0);
136 
137  const float start = static_cast<float>(randomGenHue(gen));
138  for(size_t i = 0; i < colorCount; ++i) {
139  const float added = start + static_cast<float>(i) / static_cast<float>(colorCount) * 0.7f;
140  float whole, fractional;
141  fractional = std::modf(added, &whole);
142  ci::ColorA r = cinder::hsvToRgb(glm::vec4(fractional, 1, randomGenValue(gen), 1));
143  outNewColors.emplace_back(r.r, r.g, r.b, 1);
144  }
145  }
