154  if(keycode >= ci::app::KeyEvent::KEY_a && keycode <= ci::app::KeyEvent::KEY_z) {
155  result += std::toupper('a' + static_cast<char>(keycode - ci::app::KeyEvent::KEY_a));
156  } else if(keycode >= ci::app::KeyEvent::KEY_0 && keycode <= ci::app::KeyEvent::KEY_9) {
157  result += '0' + static_cast<char>(keycode - ci::app::KeyEvent::KEY_0);
158  }
159  return result;
160  }
